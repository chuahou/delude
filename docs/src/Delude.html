module Delude (
              -- * Rewritten functions
              -- ** Folds and traversals
                foldr1
              , foldl1
              , foldr1'
              , foldl1'
              , maximum
              , minimum
              , maximum'
              , minimum'
              , sum
              , product
              -- * Non-@Prelude@ reexports
              -- ** 'Data.Foldable'
              , module Data.Foldable
              -- * Original @Prelude@ reexports
              , module Prelude
              ) where
